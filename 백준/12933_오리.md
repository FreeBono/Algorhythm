```python
import sys
sys.stdin = open('input.txt')

a = list(input())
last = []
cnt = 0
flag = 0
while True:
    ls = [0, 0, 0, 0, 0]
    if a.count(0) == len(a):
        break
    if a.count(0) == len(a)-5:
        for p in range(len(a)):
            if a[p] != 0:
                last.append(a[p])
        if last != ['q','u','a','c','k']:
            flag = 1
            break
    for i in range(len(a)):



        if a[i] == 'q' and ls[0] == 0:
            ls[0] = i+1
        elif a[i] == 'u' and ls[0] != 0 and ls[1] == 0:
            ls[1] = i+1
        elif a[i] == 'a' and ls[1] != 0 and ls[2] == 0:
            ls[2] = i+1
        elif a[i] == 'c' and ls[2] != 0 and ls[3] == 0:
            ls[3] = i+1
        elif a[i] == 'k' and ls[2] != 0 and ls[4] == 0:
            ls[4] = i+1

        if ls.count(0) == 0 and i != 0:
            for j in ls:
                a[j-1] = 0
            ls = [0,0,0,0,0]

    cnt += 1
if flag == 0:
    print(cnt)
else:
    print(-1)
```
