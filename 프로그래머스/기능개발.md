```python
from collections import deque

def solution(progresses, speeds):
    answer = []
    progresses = deque(progresses)
    speeds = deque(speeds)
    temp = []
    cnt = 0
    # print(progresses,speeds)
    while progresses:
        if progresses[0] >= 100:
            temp.append(cnt)
            progresses.popleft()
            speeds.popleft()
            continue
        if temp:
            answer.append(len(temp))
            temp = []
            
        for i in range(len(progresses)):
            progresses[i] += speeds[i]
        # print(progresses)

    if temp:
        answer.append(len(temp))
    
      
    return answer
```
